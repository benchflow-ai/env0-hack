curl Examples

# List documents
curl http://127.0.0.1:9004/v1/documents

# Search by title
curl "http://127.0.0.1:9004/v1/documents?q=Aurora"

# Read a document
curl http://127.0.0.1:9004/v1/documents/{{ sample_doc_id }}

# Create a document
curl -X POST http://127.0.0.1:9004/v1/documents \
  -H 'Content-Type: application/json' \
  -d '{"title":"Draft Summary"}'

# Insert text
curl -X POST http://127.0.0.1:9004/v1/documents/{{ sample_doc_id }}:batchUpdate \
  -H 'Content-Type: application/json' \
  -d '{"requests":[{"insertText":{"location":{"index":1},"text":"Hello\n"}}]}'

# Get environment state
curl http://127.0.0.1:9004/_admin/state

# Reset to seed
curl -X POST http://127.0.0.1:9004/_admin/reset
